## Sources

set(SOURCES
  main.c
  )

##------------------------------------------------------------------------------
## Build target
##

add_executable(wadtool ${SOURCES})

# Generate kex.wad
add_custom_target(kexwad ALL
  DEPENDS wadtool
  COMMAND wadtool "${CMAKE_SOURCE_DIR}/distrib/kex" "${CMAKE_BINARY_DIR}/kex.wad"
  )

##------------------------------------------------------------------------------
## Install target
##

# Install kex.wad
if (NOT WIN32)
  install(FILES ${CMAKE_BINARY_DIR}/kex.wad DESTINATION share/games/doom64ex)
endif ()
